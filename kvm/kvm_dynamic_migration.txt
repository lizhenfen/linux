1.配置nfs
	vi /etc/exports
	/vm *(rw,sync,no_root_squash)
2.你懂得
	service nfsserver restart
3.你懂的
	exportfs -av
4.你懂的，迁移到对端
	ssh迁移:
		virsh list --all
		virsh migrate vdisk --live qemu+ssh://目标ip地址:/system --unsafe
	tcp迁移:
		#libvirt默认不支持tcp，修改配置文件支持tcp(/etc/sysconfig/libvirtd ):
			LIBVIRTD_ARGS="--listen"
			listen_tls=0
			listen_tcp=1
			auth_tcp="none"
		修改防火墙:
			-A INPUT -p tcp -m multiport --ports 16509 -m comment --comment "libvirt" -j ACCEPT
		重启 libvirtd 服务
		测试tcp连接：	
			source host:   virsh -c qemu+tcp://nova@compute02/system
			dest   host:   virsh -c qemu+tcp://nova@compute02/system
