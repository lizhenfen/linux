PS1：
    默认: \u@\h \W: 用户名,主机名, 当前目录
    \w: 绝对路径
    \W：相对路径
    \u: 用户名
    \h: 主机名
    
PS2：
    二级目录的默认提示符
PS3:
    定制shell脚本的select的提示符
PS4:
    定义 set -x 调试,的前缀
    export PS4='$0.$LINENO+ '  //$LINENO+ 表示当前的行号
PROMPT_COMMAND:
    在执行PS1之前,先执行此变量中的内容
    export PROMPT_COMMAND="echo -n [$(date +%k:%m:%S)]"
设置提示符的颜色：
    export PS1="\e[0;34m[\u@\h \W] \e[m "

其中 \e[ ——表示颜色的提示的起始位置
x;ym ——表示颜色的代码。颜色的代码在下面有说明
\e[m ——表示代颜色的提示的结束位置
